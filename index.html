CREATE TABLE IF NOT EXISTS usuarios ( id TEXT PRIMARY KEY, tipo TEXT, nome TEXT, latitude FLOAT, longitude FLOAT, online BOOLEAN DEFAULT false, carro TEXT, criado_em TIMESTAMP DEFAULT NOW() ); CREATE TABLE IF NOT EXISTS corridas ( id TEXT PRIMARY KEY, passageiro_id TEXT, motorista_id TEXT, passageiro_nome TEXT, motorista_nome TEXT, motorista_carro TEXT, origem TEXT, origem_lat FLOAT, origem_lng FLOAT, destino TEXT, tipo TEXT, status TEXT DEFAULT 'solicitado', preco DECIMAL(10,2), criado_em TIMESTAMP DEFAULT NOW(), aceito_em TIMESTAMP, chegando_em TIMESTAMP, iniciado_em TIMESTAMP, finalizado_em TIMESTAMP );
